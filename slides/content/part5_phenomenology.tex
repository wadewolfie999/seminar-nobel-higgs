\begin{frame}{Is it the Higgs or a Higgs?}
  \centering
  \Large
  Many models predict extended Higgs sectors:
  
  \vspace{0.5em}
  \Large
  Two-Higgs-doublet models, SUSY Higgs, singlet portals, \dots
\end{frame}

\begin{frame}{Precision Higgs phenomenology}
  \centering
  \Large
  Measure couplings to $W,Z$, fermions, and itself. \\
  \vspace{0.5em}
  \Large
  Look for small deviations from SM predictions.
\end{frame}

\begin{frame}{Higgs as a portal}
  \centering
  \Large
  Higgs couplings may connect the visible sector \\
  to dark matter or hidden sectors.

  \vspace{0.5em}
  \Large
  Invisible decays, exotic decays, modified branching ratios.
\end{frame}

\begin{frame}{Open questions}
  \centering
  \Large
  Why is the Higgs so light? \\
  Why this pattern of Yukawa couplings? \\
  Is electroweak symmetry breaking unique?
\end{frame}
