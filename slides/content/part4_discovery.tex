\begin{frame}{Discovery: Does the Higgs show up?}
  \begin{itemize}
    \item Where would a Higgs leave a clear bump?
    \item Can we spot a narrow signal over huge backgrounds?
    \item Which channels balance sensitivity and precision?
  \end{itemize}
\end{frame}

\begin{frame}{A New Boson at 125 GeV}
  \begin{itemize}
    \item ATLAS and CMS both see excesses near 125~GeV.
    \item Combined channels exceed $5\sigma$ significance.
    \item Properties align with a scalar carrying Higgs-like couplings.
  \end{itemize}
\end{frame}
