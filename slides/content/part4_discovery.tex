\sectiondivider{Discovery at the LHC}{Tracking a 125 GeV resonance in clean final states.}

\begin{frame}{A short history of the hunt}
  \slidephrase{Key waypoints on the Higgs road.}
  \slidegap
  \begin{itemize}
    \item 1964: Higgs mechanism proposed to hide symmetry while giving mass.
    \item 1980s--2000s: Electroweak precision tests tightened the allowed mass window.
    \item 2010s: The LHC switched on, turning prediction into experimental chase.
  \end{itemize}
\end{frame}

\begin{frame}{Discovery: Does the Higgs show up?}
  \slidephrase{Where would a Higgs leave a clear bump?}
  \slidesubphrase{Can we see a narrow signal over huge backgrounds?}
  \slidesubphrase{Focus on clean final states with precise mass reconstruction.}
\end{frame}
