\begin{cleanframe}
  \centering
  \Huge
  What if the symmetry is hidden, not broken by hand?
\end{cleanframe}

\begin{frame}{Symmetry in the equations, not in the vacuum}
  \centering
  \Large
  Keep the Lagrangian symmetric. \\
  Let the vacuum pick a direction.
  
  \vspace{1em}
  \Large
  \emph{Spontaneous symmetry breaking.}
\end{frame}

\begin{frame}{The Higgs field idea}
  \centering
  \Large
  Introduce a complex scalar doublet $\Phi$ with potential
  \[
    V(\Phi) = \mu^2 \Phi^\dagger \Phi + \lambda (\Phi^\dagger \Phi)^2
  \]
  \vspace{0.5em}
  Choose $\mu^2 < 0$ \textrightarrow\ the vacuum is not at $\Phi = 0$.
\end{frame}
