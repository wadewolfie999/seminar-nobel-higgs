\sectiondivider{Theory: The problem of mass}{Gauge symmetry wants light fields; experiments do not.}

\begin{frame}{Why is mass a problem?}
  \slidephrase{Gauge theories love massless fields.}
  \slidesubphrase{But nature is full of massive particles.}
  \slidesubphrase{Naive mass terms seem to break our symmetries.}
\end{frame}

\begin{frame}{Gauge bosons and symmetry}
  \slidephrase{Yang--Mills dynamics keep the gauge symmetry intact.}
  \slideeq{\mathcal{L}_{\text{YM}} = -\frac{1}{4} F_{\mu\nu}^a F^{a\,\mu\nu}}
  \slidesubphrase{Gauge symmetry protects vector bosons from a mass term.}
\end{frame}

\begin{frame}{The forbidden mass term}
  \slidephrase{Writing a mass by hand breaks the symmetry.}
  \slideeq{\mathcal{L}_{\text{Proca}} = \frac{1}{2} m^2 A_\mu A^\mu}
  \slidesubphrase{The Proca term is not gauge invariant.}
\end{frame}

\begin{frame}{Fermion mass trouble}
  \slidephrase{A bare Dirac mass clashes with electroweak charges.}
  \slideeq{\mathcal{L}_{\text{Dirac}} = -m \overline{\psi}\, \psi}
  \slidesubphrase{Left and right chiralities transform differently under $SU(2)\times U(1)$.}
\end{frame}

\begin{frame}{The conflict}
  \slidetension{Gauge symmetry forbids naive mass.}{Experiments demand massive $W$, $Z$, and fermions.}
  \slidesubphrase{We need a new idea that keeps the symmetry but generates mass.}
\end{frame}

\begin{frame}{A clue: symmetry can hide itself}
  \slidephrase{Equations stay symmetric.}
  \slidesubphrase{The ground state can be asymmetric.}
  \slidesubphrase{Next: a simple scalar field that hides symmetry.}
\end{frame}
