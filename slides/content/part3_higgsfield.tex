\begin{frame}{Choosing a vacuum}
  \centering
  \Large
  Choose one vacuum direction:
  \[
    \Phi(x) =
    \frac{1}{\sqrt{2}}
    \begin{pmatrix}
      0 \\
      v + h(x)
    \end{pmatrix}
  \]
  
  \vspace{0.5em}
  \Large
  The symmetry of the equations survives; \\
  the symmetry of the vacuum does not.
\end{frame}

\begin{frame}{Gauge boson masses from the vacuum}
  \centering
  \Large
  Gauge fields eat Goldstone modes and become massive:
  \[
    m_W = \tfrac{1}{2} g v,
    \qquad
    m_Z = \tfrac{1}{2} \sqrt{g^2 + g'^2}\, v
  \]
  
  \vspace{0.5em}
  \Large
  Mass comes from interaction with an always-on field.
\end{frame}

\begin{frame}{The Higgs boson}
  \centering
  \Large
  $h(x)$ is a real scalar excitation of the Higgs field. \\
  \vspace{0.5em}
  \Large
  A vibration of the vacuum itself.
\end{frame}
