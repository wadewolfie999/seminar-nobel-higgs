\sectiondivider{The Higgs field}{A permeating field that lends mass to particles.}

\begin{frame}{A field that fills space}
  \slidephrase{Introduce the Higgs as an $SU(2)$ doublet.}
  \slidesubphrase{A field that is nonzero everywhere in the vacuum.}
  \slidesubphrase{Its presence reshapes the behavior of gauge and matter fields.}
\end{frame}

\begin{frame}{Choosing the Vacuum}
  \slidephrase{We choose a vacuum $\langle \phi \rangle = (0, v/\sqrt{2})^{\mathrm T}$.}
  \slidesubphrase{One direction in the doublet acquires a constant value.}
  \slidesubphrase{Here we visualize this \alert{vacuum expectation value} with the Higgs potential.}
\end{frame}

\begin{frame}{Gauge Boson Masses}
  \slidephrase{Gauge fields absorb Goldstone modes and become massive.}
  \slideeq{m_W = \tfrac{g v}{2}}
  \slidesubphrase{The $Z$ acquires a matching mass set by $g$ and $g'$.}
  \slidesubphrase{One vacuum scale $v$ anchors the vector spectrum.}
\end{frame}

\begin{frame}{Fermion Masses}
  \slidephrase{Yukawa interactions turn the VEV into fermion masses.}
  \slidesubphrase{$-y_f\, \bar{\psi}_L \Phi \psi_R + \text{h.c.}$ couples chiral partners.}
  \slideeq{m_f = \dfrac{y_f v}{\sqrt{2}}}
\end{frame}

\begin{frame}{One Scalar Remains}
  \slidephrase{Goldstone modes are eaten by the gauge fields.}
  \slidesubphrase{One physical scalar excitation is left.}
  \slidesubphrase{This is the Higgs boson we aim to discover.}
\end{frame}

\begin{frame}{Why This Matters Experimentally}
  \slidephrase{Masses and couplings are now calculable.}
  \slidesubphrase{Parameters $v$, $g$, and $y_f$ set cross sections and branching ratios.}
  \slidesubphrase{Predicted rates single out clean final states: $\gamma\gamma$, $ZZ\!\to\!4\ell$, $WW$, $b\bar b$.}
  \slidesubphrase{Searches pursue a specific pattern rather than guessing channels.}
\end{frame}

\begin{frame}{Can we see the Higgs?}
  \slidephrase{A scalar with predicted couplings and a narrow width.}
  \slidesubphrase{Golden peaks in $ZZ\!\to\!4\ell$ and $\gamma\gamma$ pin down the mass.}
  \slidesubphrase{Cross-checks in $WW$, $b\bar b$, and $\tau\tau$ fill out the coupling map.}
  \slidesubphrase{ATLAS and CMS know the signatures and how tall a bump to expect.}
\end{frame}
