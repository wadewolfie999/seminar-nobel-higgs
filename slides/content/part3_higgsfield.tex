\begin{frame}{A field that fills space}
\centering
Higgs field: complex SU(2) doublet $\Phi = (\phi^+,\,\phi^0)^T$.
Introduced to endow gauge bosons with mass while keeping gauge symmetry.
\end{frame}

\begin{frame}{Choosing the Vacuum}
\centering
$\langle \Phi \rangle = \begin{pmatrix} 0 \\ v/\sqrt{2} \end{pmatrix}$ selects one direction.
\vspace{0.6em}

\vspace{0.4em}
Symmetry of the laws persists; the vacuum hides it.
\end{frame}

\begin{frame}{Gauge Boson Masses}
\centering
Gauge fields absorb Goldstone modes and become massive.
\[
 m_W = \tfrac{g v}{2}, \qquad m_Z = \tfrac{v}{2}\sqrt{g^2 + g'^2}
\]
Mass arises from interacting with the constant Higgs background.
\end{frame}

\begin{frame}{Fermion Masses}
\centering
Yukawa term: $-y_f\, \bar{\psi}_L \Phi \psi_R + \text{h.c.}$
VEV turns Yukawa coupling into mass.
\[
 m_f = \dfrac{y_f v}{\sqrt{2}}
\]
\end{frame}

\begin{frame}{One Scalar Remains}
\centering
After symmetry breaking, three Goldstone modes are eaten.
A single physical excitation survives: the Higgs boson $h$.
\end{frame}

\begin{frame}{Why This Matters Experimentally}
\centering
Predicts a new scalar with couplings set by mass relations.
Its mass and decay patterns define experimental searches.
\end{frame}

\begin{frame}{Can we see the Higgs?}
  \centering
  A scalar with predicted couplings.\\[0.7em]
  Specific rates into ZZ, $\gamma\gamma$, and other channels.\\[0.7em]
  ATLAS and CMS can test this.
\end{frame}
