% Higgs seminar slide layout macros

% Vertical gap macro for consistent rhythm
\newcommand{\slidegap}{\vspace{1.05em plus 0.35em minus 0.3em}}

% Single centered phrase, tuned for minimal-text slides
\newcommand{\slidephrase}[1]{%
  \par\slidegap
  \begin{center}{\Large\bfseries\color{black!90} #1}\end{center}%
  \slidegap\par
}

% Smaller sub-phrase (e.g. second or third line)
\newcommand{\slidesubphrase}[1]{%
  \par\slidegap
  \begin{center}{\small\color{gray!60} #1}\end{center}%
  \par
}

% Equation block with symmetric spacing
\newcommand{\slideeq}[1]{%
  \par\slidegap
  {\setlength{\abovedisplayskip}{8pt plus 3pt minus 2pt}%
   \setlength{\belowdisplayskip}{8pt plus 3pt minus 2pt}%
   \setlength{\abovedisplayshortskip}{6pt plus 2pt minus 2pt}%
   \setlength{\belowdisplayshortskip}{6pt plus 2pt minus 2pt}%
   \[
     #1
   \]}
  \slidegap\par
}

% Standardized figure block (e.g. discovery plots)
\newcommand{\slidefigure}[2][0.78]{%
  \par\vspace{0.55em plus 0.25em minus 0.2em}
  \begin{center}
    \includegraphics[width=#1\linewidth]{#2}
  \end{center}
  \vspace{0.75em plus 0.35em minus 0.25em}\par
}

% Two-line “tension” structure: statement + consequence
\newcommand{\slidetension}[2]{%
  \par\slidegap
  \begin{center}
    {\Large\bfseries\color{black!90} #1}\\[0.7em]
    {\small\color{gray!60} #2}
  \end{center}
  \slidegap\par
}
