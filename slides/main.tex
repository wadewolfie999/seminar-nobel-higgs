% Higgs seminar slide layout macros

% Vertical gap macro for consistent rhythm
\newcommand{\slidegap}{\vspace{0.6em plus 0.25em minus 0.2em}}

% Single centered phrase, tuned for minimal-text slides
\newcommand{\slidephrase}[1]{%
  \par\slidegap
  \begin{center}\large #1\end{center}%
  \slidegap\par
}

% Smaller sub-phrase (e.g. second or third line)
\newcommand{\slidesubphrase}[1]{%
  \par\slidegap
  \begin{center}\normalsize #1\end{center}%
  \par
}

% Equation block with symmetric spacing
\newcommand{\slideeq}[1]{%
  \par\slidegap
  {\setlength{\abovedisplayskip}{6pt plus 3pt minus 2pt}%
   \setlength{\belowdisplayskip}{6pt plus 3pt minus 2pt}%
   \[
     #1
   \]}
  \slidegap\par
}

% Standardized figure block (e.g. discovery plots)
\newcommand{\slidefigure}[2][0.82]{%
  \par\vspace{0.35em plus 0.25em minus 0.15em}
  \begin{center}
    \includegraphics[width=#1\linewidth]{#2}
  \end{center}
  \vspace{0.5em plus 0.3em minus 0.2em}\par
}

% Two-line “tension” structure: statement + consequence
\newcommand{\slidetension}[2]{%
  \par\slidegap
  \begin{center}
    \large #1\\[0.65em]
    \normalsize #2
  \end{center}
  \slidegap\par
}

\newcommand{\sectiondivider}[2]{%
  \begin{frame}[plain]

      % Increase spacing between lines
      \draw[gray!45, line width=0.6pt] (-0.55\paperwidth,0.85) -- (0.55\paperwidth,0.85);  % Adjusted line position
      \draw[gray!45, line width=0.6pt] (-0.55\paperwidth,-0.85) -- (0.55\paperwidth,-0.85);  % Adjusted line position
      \fill[gray!10, rounded corners=4pt] (-0.5\paperwidth,-0.5) rectangle (0.6\paperwidth,0.6);  % Adjusted box height
      \draw[myblue, line width=0.7pt, rounded corners=4pt] (-0.5\paperwidth,-0.5) rectangle (0.6\paperwidth,0.6);  % Adjusted box height

      % Adjust title and subtitle positions to prevent overlap
      \node[align=center] at (0,0.25) {\Large\bfseries #1};  % Adjusted title position (increased vertical space)
      \node[align=center,color=gray!70] at (0,-0.25) {\small #2};  % Adjusted subtitle position (increased vertical space)
    \end{tikzpicture}
    \vfill
  \end{frame}
}
